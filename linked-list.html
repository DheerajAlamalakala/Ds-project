struct Node {
    int data;
    struct Node* next;
};

struct Node* head = NULL;

void insertHead(int val) {
    struct Node* n = malloc(sizeof(struct Node));
    n->data = val;
    n->next = head;
    head = n;
}

void insertTail(int val) {
    struct Node* n = malloc(sizeof(struct Node));
    n->data = val;
    n->next = NULL;
    if (head == NULL) { head = n; return; }
    struct Node* cur = head;
    while (cur->next != NULL) cur = cur->next;
    cur->next = n;
}

void deleteValue(int val) {
    if (head == NULL) return;
    if (head->data == val) { head = head->next; return; }
    struct Node* cur = head;
    while (cur->next && cur->next->data != val) cur = cur->next;
    if (cur->next) cur->next = cur->next->next;
}

How this works

Unlike an array, a linked list has no fixed memory block — each node is allocated separately and holds a pointer, next, to wherever the following node happens to live in memory. head is the only fixed reference point: lose it, and the rest of the list becomes unreachable.

Insert at head is O(1): the new node's next is set to the current head, then head is updated to point at the new node — no traversal needed.

Insert at tail and delete both require walking the list one node at a time with a cur pointer until the right spot is found, since there's no shortcut to "the last node" or "the node before X" the way there is with array indices.
