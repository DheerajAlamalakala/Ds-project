struct Node {
    int data;
    struct Node* prev;
    struct Node* next;
};

void insertTail(struct Node** head, int val) {
    struct Node* newNode = malloc(sizeof(struct Node));
    newNode->data = val;
    newNode->next = NULL;
    
    if (*head == NULL) {
        newNode->prev = NULL;
        *head = newNode;
        return;
    }
    
    struct Node* temp = *head;
    while (temp->next != NULL) temp = temp->next;
    
    temp->next = newNode;
    newNode->prev = temp;
}
