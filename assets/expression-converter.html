void infixToPostfix(char* exp) {
    for(int i=0; exp[i]; i++) {
        if(isalnum(exp[i])) 
            printf("%c", exp[i]);
        else if(exp[i] == '(') 
            push(exp[i]);
        else if(exp[i] == ')') {
            while(peek() != '(') printf("%c", pop());
            pop();
        } else {
            while(prec(exp[i]) <= prec(peek())) 
                printf("%c", pop());
            push(exp[i]);
        }
    }
    while(top != -1) printf("%c", pop());
}
